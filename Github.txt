# CHECK CHANGES
git status
git diff
git diff --staged

# COMMIT HISTORY
git log --oneline

# COMMIT CHANGES
git show <commit-id>

# FILE HISTORY + CHANGES
git log -p -- 6_methods/A_1_basic.java

# VIEW OLD CODE (NO CHANGE)
git show <commit-id>:6_methods/A_1_basic.java

# REMOVE UNCOMMITTED CHANGES
git restore 6_methods/A_1_basic.java

# RESTORE FILE TO OLD COMMIT
git log --oneline
git checkout <commit-id> -- 6_methods/A_1_basic.java
git add 6_methods/A_1_basic.java
git commit -m "Reverted A_1_basic.java"
git push

git show <commit-id>

# MEMORY FORMULA
New Project:
git init → git add → git commit → git push

Update Code:
git add → git commit → git push

Check History:
git log --oneline → git show <commit-id>

Restore Old Version:
git log --oneline → git checkout <commit-id> -- <file> → git commit → git push


git init → only once per project.
git add . → all files.
git add file.java → specific file.
git show → see changes.
git restore → discard uncommitted changes.
git checkout <commit-id> -- file → restore old committed version of a file. 🚀
